webpack的使用

怎样用webpack-dev-server来部署js

2021-11-08

Written by: tdtc

安装webpack

npm install -g webpack-dev-server
npm install --save webpack-cli

pack info

package.json

{
  "name": "redux-book-started",
  "version": "1.0.0",
  "description": "Sample Redux Application",
  "main": "app.js",
  "scripts": {
    "start": "webpack-dev-server --hot --progress  --port 8181"
  },
  "author": "Boris Dinkevich & Ilya Gelman",
  "license": "MIT",
  "dependencies": {
    "jquery": "3.6.0",
    "redux": "4.1.2"
  },
  "devDependencies": {
    "webpack": "5.64.1",
    "webpack-cli": "^4.9.1",
    "webpack-dev-server": "4.5.0"
  }
}

config info

webpack.config.js

module.exports = {
  mode: 'development',
  entry: './app.js',
  output: {
    filename: './app.js'
  }
};

run

npm install
npm start
