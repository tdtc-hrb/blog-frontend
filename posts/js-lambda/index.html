JS的lambda表达式

2020-04-10

Written by: tdtc

Before rewriting

function selectedReddit(state = 'user/8', action){
  switch (action.type) {
    case SELECT_REDDIT:
      return action.reddit
    default:
      return state
  }
}

After rewriting

const selectedReddit = (state = 'user/8', action) => {
  switch (action.type) {
    case SELECT_REDDIT:
      return action.reddit
    default:
      return state
  }
}
